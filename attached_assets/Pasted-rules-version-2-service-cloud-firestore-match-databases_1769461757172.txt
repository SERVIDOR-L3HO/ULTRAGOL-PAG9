rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Usuarios - Cualquier usuario autenticado puede leer
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create, update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // NUEVO: Contador de Presencia (Real-time Presence)
    match /status/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // NUEVO: Estadísticas globales (Contador Online)
    match /stats/online_users {
      allow read: if true;
      allow write: if false; // Solo actualizable via Admin SDK o Cloud Functions
    }

    // Códigos promocionales - MÁXIMO 14 DÍAS (Tu regla original)
    match /promoCodes/{codeId} {
      allow read: if request.auth != null;
      
      allow update: if request.auth != null &&
                      resource.data.isActive == true &&
                      (resource.data.maxUses == 0 || resource.data.currentUses < resource.data.maxUses) &&
                      request.resource.data.diff(resource.data).affectedKeys()
                        .hasOnly(['currentUses', 'lastUsedAt']) &&
                      request.resource.data.currentUses == resource.data.currentUses + 1;
      
      allow create: if request.auth != null && 
                      request.resource.data.durationDays <= 14;
      
      allow delete: if request.auth != null;
      
      allow update: if request.auth != null && 
                      request.resource.data.durationDays <= 14;
    }
    
    // Chat en vivo (Tu regla original)
    match /liveChatMessages/{messageId} {
      allow read: if true;
      allow create: if request.auth != null 
                    && request.resource.data.userId == request.auth.uid;
      allow update, delete: if request.auth != null 
                    && request.auth.uid == resource.data.userId;
    }
    
    // Equipos
    match /teams/{teamId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Streams
    match /streams/{streamId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null 
                            && request.auth.uid == resource.data.userId;
    }
    
    // Comentarios
    match /comments/{commentId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null 
                            && request.auth.uid == resource.data.userId;
    }
    
    // Notificaciones
    match /notifications/{notificationId} {
      allow read: if request.auth != null 
                  && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null 
                            && request.auth.uid == resource.data.userId;
    }
    
    // Donaciones
    match /donations/{donationId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if false;
      allow delete: if false;
    }
    
    // Links de partidos
    match /match_links/{linkId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null 
                            && request.auth.uid == resource.data.userId;
    }
    
    // Seguridad global
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
